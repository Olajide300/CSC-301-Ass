# Binary Search in Python

values = [1, 3, 5, 7, 9, 11, 13]
target = 9

def binary_search(arr, target):
    low = 0
    high = len(arr) - 1

    while low <= high:
        mid = (low + high) // 2

        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            low = mid + 1
        else:
            high = mid - 1

    return -1   # target not found

# Run the search
index = binary_search(values, target)

if index != -1:
    print(f"Target {target} found at index {index}.")
else:
    print("Target not found.")


Time & Space Complexity

Linear Search
	Time Complexity: O(n)
  (Worst case: check every element)
	Space Complexity: O(1)
  (No extra memory used)

Binary Search
 	Time Complexity: O(log n)
  (Each step divides the array by 2)
	Space Complexity:
  Iterative version: O(1)
  Recursive version: O(log n) [because of call stack]
